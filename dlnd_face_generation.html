Preprocess the Data¶

Since the project's main focus is on building the GANs, we'll preprocess the data for you. The values of the MNIST and CelebA dataset will be in the range of -0.5 to 0.5 of 28x28 dimensional images. The CelebA images will be cropped to remove parts of the image that don't include a face, then resized down to 28x28.

The MNIST images are black and white images with a single color channel while the CelebA images have 3 color channels (RGB color channel).

def discriminator(images, reuse=False, alpha=0.2):
    """
    Create the discriminator network
    :param images: Tensor of input image(s)
    :param reuse: Boolean if the weights should be reused
    :return: Tuple of (tensor output of the discriminator, tensor logits of the discriminator)
    """
    # TODO: Implement Function
    with tf.variable_scope('discriminator', reuse=reuse):
        # Conv 1 layer - 14x14x64
        conv1 = tf.layers.conv2d(images, 64, 4, strides=2, padding='same')
        conv1 = tf.maximum(alpha * conv1, conv1)
        
        # Conv 2 layer - 7x7x128
        conv2 = tf.layers.conv2d(conv1, 128, 4, strides=2, padding='same')
        conv2 = tf.layers.batch_normalization(conv2, training=True)
        conv2 = tf.maximum(alpha * conv2, conv2)
        
        # Conv 3 layer - 4x4x256
        conv3 = tf.layers.conv2d(conv2, 256, 4, strides=2, padding='same')
        conv3 = tf.layers.batch_normalization(conv3, training=True)
        conv3 = tf.maximum(alpha * conv3, conv3)
        
        # Dense
        dense = tf.contrib.layers.flatten(conv3)
        logits = tf.layers.dense(dense, 1)
        output = tf.sigmoid(logits)

    return output, logits

Generator¶

Implement generator to generate an image using z. This function should be able to reuse the variables in the neural network. Use tf.variable_scope with a scope name of "generator" to allow the variables to be reused. The function should return the generated 28 x 28 x out_channel_dim images.

def generator(z, out_channel_dim, is_train=True, alpha=0.2):
    """
    Create the generator network
    :param z: Input z
    :param out_channel_dim: The number of channels in the output image
    :param is_train: Boolean if generator is being used for training
    :return: The tensor output of the generator
    """
    # TODO: Implement Function
    with tf.variable_scope('generator', reuse=not is_train):
        # Expand and reshape
        inputs = tf.layers.dense(z, 4*4*512, use_bias=False)
        inputs = tf.reshape(inputs, (-1, 4, 4, 512))
        inputs = tf.layers.batch_normalization(inputs, training=is_train)
        inputs = tf.maximum(alpha * inputs, inputs)
        
        # Conv 1 - 7x7x256
        conv1 = tf.layers.conv2d_transpose(inputs, 256, 4, strides=1, padding='valid', use_bias=False)
        conv1 = tf.layers.batch_normalization(conv1, training=is_train)
        conv1 = tf.maximum(alpha * conv1, conv1)
        
        # Conv 2 - 14x14x128
        conv2 = tf.layers.conv2d_transpose(conv1, 128, 4, strides=2, padding='same', use_bias=False)
        conv2 = tf.layers.batch_normalization(conv2, training=is_train)
        conv2 = tf.maximum(alpha * conv2, conv2)
        
        # Conv 3 - 28x28x3
        conv3 = tf.layers.conv2d_transpose(conv2, out_channel_dim, 4, strides=2, padding='same', use_bias=False)
        output = tf.tanh(conv3) / 2
    
    return output

def model_loss(input_real, input_z, out_channel_dim):
    """
    Get the loss for the discriminator and generator
    :param input_real: Images from the real dataset
    :param input_z: Z input
    :param out_channel_dim: The number of channels in the output image
    :return: A tuple of (discriminator loss, generator loss)
    """
    # TODO: Implement Function
    g_output = generator(input_z, out_channel_dim, is_train=True)
    d_real_output, d_real_logits = discriminator(input_real, reuse=False)
    d_fake_output, d_fake_logits = discriminator(g_output, reuse=True)
    
    d_real_loss = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(logits=d_real_logits, 
                                                                         labels=tf.ones_like(d_real_output)))
    d_fake_loss = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(logits=d_fake_logits, 
                                                                         labels=tf.zeros_like(d_fake_output)))
    d_loss = d_real_loss + d_fake_loss
    
    g_loss = tf.reduce_mean(tf.nn.sigmoid_cross_entropy_with_logits(logits=d_fake_logits, 
                                                                         labels=tf.ones_like(d_fake_output)))
    
    return d_loss, g_loss

def model_opt(d_loss, g_loss, learning_rate, beta1):
    """
    Get optimization operations
    :param d_loss: Discriminator loss Tensor
    :param g_loss: Generator loss Tensor
    :param learning_rate: Learning Rate Placeholder
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :return: A tuple of (discriminator training operation, generator training operation)
    """
    # TODO: Implement Function
    t_var = tf.trainable_variables()
    d_var = [var for var in t_var if var.name.startswith('discriminator')]
    g_var = [var for var in t_var if var.name.startswith('generator')]

    d_opt = tf.train.AdamOptimizer(learning_rate=learning_rate, beta1=beta1).minimize(d_loss, var_list=d_var)
    g_opt = tf.train.AdamOptimizer(learning_rate=learning_rate, beta1=beta1).minimize(g_loss, var_list=g_var)
    
    return d_opt, g_opt

def train(epoch_count, batch_size, z_dim, learning_rate, beta1, get_batches, data_shape, data_image_mode):
    """
    Train the GAN
    :param epoch_count: Number of epochs
    :param batch_size: Batch Size
    :param z_dim: Z dimension
    :param learning_rate: Learning Rate
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :param get_batches: Function to get batches
    :param data_shape: Shape of the data
    :param data_image_mode: The image mode to use for images ("RGB" or "L")
    """
    # TODO: Build Model
    # Placeholders
    real_inputs, z_data, learn_rate = model_inputs(data_shape[1], data_shape[2], data_shape[3], z_dim)
    
    # Build computational graph
    d_loss, g_loss = model_loss(real_inputs, z_data, data_shape[3])
    d_opt, g_opt = model_opt(d_loss, g_loss, learn_rate, beta1)
    
    # iterations per epoch
    iter_per_epoch = data_shape[0] // batch_size
    show_stats = iter_per_epoch // 10
    show_faces = iter_per_epoch // 4
    
    # Training loop
    with tf.Session() as sess:
        sess.run(tf.global_variables_initializer())
        for epoch_i in range(epoch_count):
            step = 0
            for batch_images in get_batches(batch_size):
                # TODO: Train Model
                step += 1
                # define random vector to feed into generator
                z_vector = np.random.uniform(-1.0, 1.0, size=(batch_size, z_dim))
                
                # run optimizers
                _ = sess.run(d_opt, feed_dict={real_inputs: batch_images, 
                                               z_data: z_vector, 
                                               learn_rate: learning_rate})
                _ = sess.run(g_opt, feed_dict={real_inputs: batch_images, 
                                               z_data: z_vector, 
                                               learn_rate: learning_rate})
                
                # calculate and display loss
                if step % show_stats == 0:
                    d_loss_stat = sess.run(d_loss, feed_dict={real_inputs: batch_images, 
                                                              z_data: z_vector, 
                                                              learn_rate: learning_rate})
                    g_loss_stat = sess.run(g_loss, feed_dict={real_inputs: batch_images, 
                                                              z_data: z_vector, 
                                                              learn_rate: learning_rate})
                    print("Epoch: {}/{}...".format(epoch_i + 1, epoch_count), 
                          "Discriminator Loss: {:.4f}...".format(d_loss_stat),
                          "Generator Loss: {:.4f}...".format(g_loss_stat))
                
                # generate faces
                if step % show_faces == 0:
                    show_generator_output(sess, 10, z_data, data_shape[3], data_image_mode)
